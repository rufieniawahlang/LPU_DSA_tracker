class Solution:
    def checkPerfectNumber(self, num: int) -> bool:
        if num == 1:
            return False

        sum, i = 1, 2

        while i * i <= num:# y i*i ?
            if num % i == 0:
                sum += i

                if i * i != num:
                    sum += num // i

            i += 1

        return sum == num
